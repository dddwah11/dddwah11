每日算法-quicksort

2022/4/13

有没有既不浪费空间又可以快一点的排序算法呢？那就是快速排啦！

/**
 * @author: 61分
 * @date: 2022/4/14 0:05
 * @description:
 */
public class quickSort {
    public static void quickSort(int [] arr,int low,int high){

        int i,j,tmp,t;
        if (low>high){
            return;
        }
        i = low;
        j = high;

        tmp = arr[low];
        while (i<j){
            //        先看右边,从最右边递减

            while (tmp<=arr[j]&&i<j){
                j--;
            }
//            再看左边，递加
            while (tmp>=arr[i]&&i<j){
                i++;
            }
//            满足条件的话，交换他们的值
            if (i<j){
                t = arr[j];
                arr[j] = arr[i];
                arr[i] = t;
            }
        }
        arr[low] = arr[i];
        arr[i] = tmp;
//            之后递归调用 处理左边的
        quickSort(arr,low,j-1);
//            处理右边的
        quickSort(arr,j+1,high);


    }

    public static void main(String[] args) {
        int [] num= {10,54,21,42,78,65,48};
        quickSort(num,0,num.length-1);
        for (int i = 0; i < num.length; i++) {
            System.out.println(num[i]);

        }
    }
}
